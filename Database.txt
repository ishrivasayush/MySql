create table employee(
id INT, 
firstname varchar(20) NOT NULL,
middlename varchar(20),
lastname varchar(20) NOT NULL,
age int NOT NULL,
salary int NOT NULL,
location varchar(20) NOT NULL Default 'Banglore'
PRIMARY KEY(id,name)
);

//AUTOINCREMENT

create table employee(
id INT AUTO_INCREMENT,

middlename varchar(20),
lastname varchar(20) NOT NULL,
age int NOT NULL,
salary int NOT NULL,
location varchar(20) NOT NULL Default 'Banglore'
PRIMARY KEY(id,name)
);
